public interface VariableContract extends Serializable {

    String getName();

    String getModule();

    Enum<?> getInfoKey();

    Enum<?> getLabelKey();

    default String getId() {
        Objects.requireNonNull(getModule(), 
	    "A variable contract must define an module which it is part of");
        
		return getModule() + "." + getName();
    }

    default String toInfoString() {
		final String ls = System.lineSeparator();
		final StringBuilder sb = new StringBuilder();
		sb.append("contract  : ").append(this.getClass().getName())
		  .append(ls)
		  .append("id        : ").append(getId())
		  .append(ls)
		  .append("name      : ").append(getName())
		  .append(ls)
		  .append("label-key : ").append((getLabelKey() != null) 
										? getLabelKey().name() 
										: "not available")
		  .append(ls)
		  .append("info-key  : ").append((getInfoKey() != null) 
										? getInfoKey().name() 
										: "not available")
		  .append(ls)
		  .toString();
    }
}
